import { Checkbox ,CheckboxGroup } from 'tailv'

# CheckboxGroup

export const checkList = ref(['A'])

<div class="">
  {checkList.value}
  <CheckboxGroup value={checkList.value} onUpdate:value={(val)=> {
    console.log(val)
    checkList.value = val
  }}> 
    <Checkbox value="A">A</Checkbox> 
    <Checkbox value="B">B</Checkbox>
  </CheckboxGroup> 
</div>


## Props

| Prop        | Type    | Default | Description                   |
| ----------- | ------- | ------- | ----------------------------- |
| **checked** | boolean | false   | whether the checkbox checked  |
| name        | string  | -       | input name                    |
| value       | any     | -       | value for group               |
| disabled    | boolean | false   | whether the checkbox disabled |

## Slots

| Name    | Description      |
| ------- | ---------------- |
| default | default children |
